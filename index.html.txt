    <script>
        const canvas = document.getElementById('meme-canvas');
        const ctx = canvas.getContext('2d');
        let currentStyle = 'red';

        // ScrollReveal con impostazioni di fallback
        try {
            ScrollReveal().reveal('.reveal', { 
                delay: 150, 
                distance: '20px',
                origin: 'bottom',
                duration: 800,
                interval: 80,
                afterReveal: function (el) { el.style.opacity = '1'; }
            });
        } catch(e) {
            console.log("ScrollReveal non caricato, forzo visibilità");
            document.querySelectorAll('.reveal').forEach(el => el.style.opacity = '1');
        }

        function scrollToGenerator() {
            document.getElementById('strumenti').scrollIntoView({ behavior: 'smooth' });
        }

        function setMemeStyle(style) {
            currentStyle = style;
            generateMeme();
        }

        function generateMeme() {
            const text = document.getElementById('meme-text').value.toUpperCase() || "BOICOTTIAMO LE TASSE";
            
            let bgColor = "#ff3333";
            let textColor = "#ffffff";
            let accentColor = "#000000";

            if(currentStyle === 'black') {
                bgColor = "#0a0a0a";
                textColor = "#ff3333";
                accentColor = "#ffffff";
            } else if(currentStyle === 'yellow') {
                bgColor = "#fbbf24";
                textColor = "#000000";
                accentColor = "#000000";
            }

            // Sfondo
            ctx.fillStyle = bgColor;
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Texture grunge simulata
            ctx.globalAlpha = 0.05;
            for(let i=0; i<3000; i++) {
                ctx.fillStyle = "#ffffff";
                ctx.fillRect(Math.random()*canvas.width, Math.random()*canvas.height, 2, 2);
            }
            ctx.globalAlpha = 1.0;

            // Cornice
            ctx.strokeStyle = textColor;
            ctx.lineWidth = 20;
            ctx.strokeRect(30, 30, canvas.width - 60, canvas.height - 60);

            // Testo
            ctx.fillStyle = textColor;
            ctx.font = "900 65px Oswald";
            ctx.textAlign = "center";
            ctx.textBaseline = "middle";
            
            const words = text.split(' ');
            let startY = (canvas.height / 2) - ((words.length - 1) * 40);
            
            words.forEach((word, index) => {
                ctx.fillText(word, canvas.width/2, startY + (index * 80));
            });

            // Firma
            ctx.fillStyle = accentColor;
            ctx.font = "700 18px Roboto";
            ctx.letterSpacing = "4px";
            ctx.fillText("RIPRENDITI IL TUO LAVORO", canvas.width / 2, canvas.height - 80);
        }

        function downloadMeme() {
            const link = document.createElement('a');
            link.download = 'boicottiamo-tasse.png';
            link.href = canvas.toDataURL();
            link.click();
        }

        // Marquee Animation
        const marquee = document.getElementById('marquee');
        let offset = 0;
        function animateMarquee() {
            offset -= 1.5;
            if (Math.abs(offset) >= marquee.scrollWidth / 2) {
                offset = 0;
            }
            marquee.style.transform = `translateX(${offset}px)`;
            requestAnimationFrame(animateMarquee);
        }

        window.onload = () => {
            generateMeme();
            animateMarquee();
        };
    </script>
